pw_calculator_particle_supercollider = {
	category = complex_drone
	swappable_data = {
		default = {
			condition_string = DRONE_JOB_TRIGGER
			building_icon = pw_building_particle_supercollider
			icon = pw_calculator_particle_supercollider
		}
		#Supercollider Brain Drone - auth_hive
		swap_type = {
			trigger = {
				exists = owner
				owner = {
					is_hive_empire = yes
					# is_wilderness_empire = no
				}
			}
			name = pw_brain_drone_particle_supercollider
			desc = pw_brain_drone_particle_supercollider
			icon = pw_brain_drone_particle_supercollider
			building_icon = pw_building_particle_supercollider
			condition_string = DRONE_JOB_TRIGGER
			weight = 50
		}
	}
	
	tags = { research }

	possible_pre_triggers = {
		has_owner = yes
		is_enslaved = no
		is_being_purged = no
		is_being_assimilated = no
		is_sapient = yes
	}

	possible_precalc = can_fill_drone_job

	possible = {
		# drone_job_check_trigger = yes
		can_think = yes
	}

	resources = {
		category = planet_physicists
		produces = {
			physics_research = 12
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					is_hive_empire = no
				}
			}
			energy = 8
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					is_hive_empire = yes
				}
			}
			minerals = 6
			energy = 3
		}
	}

	weight = {
		weight = @complex_drone_job_weight
		mult = value:job_weights_research_modifier|JOB|pw_calculator_particle_supercollider|
		modifier = {
			factor = 0.9
			exists = owner
			years_passed < 1
			owner = { has_valid_civic = civic_machine_assimilator }
			has_trait = trait_cybernetic
		}
	}

	inline_script = "jobs/automodding_priority_research"

	promotion = {
		time = @standard_promotion_time
		effect = {
			pop_amount_change_category = {
				CATEGORY = complex_drone
				AMOUNT = local_pop_amount
			}
		}
	}
}
